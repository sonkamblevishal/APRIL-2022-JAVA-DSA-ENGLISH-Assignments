//GKSTR09 Print_Range

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
      int n = scan.nextInt();
      
      for(int i=1 ; i<=n; i++)
      {
        System.out.println(i);
      }
    }
}
