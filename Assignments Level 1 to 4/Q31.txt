import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        /* Enter your code here. Read input from STDIN. Print output to STDOUT. Your class should be named Solution. */  Scanner scan = new Scanner(System.in);
    int number = scan.nextInt();
    
    


    // Outer loop 1
    // prints the first half diamond
    for (int m = 1; m <= number; m++) {

      // Inner loop 1 print whitespaces inbetween
      for (int n = 1; n <= number - m; n++) {
        System.out.print(" ");
      }

      // Inner loop 2 prints star
      for (int n = 1; n <= m * 2 - 1; n++) {
        System.out.print("*");
      }

      // Ending line after each row
      System.out.println();
    }

    // Outer loop 2
    // Prints the second half diamond
    for (int m = number - 1; m > 0; m--) {

      // Inner loop 1 print whitespaces inbetween
      for (int n = 1; n <= number - m; n++) {
        System.out.print(" ");
      }

      // Inner loop 2 print star
      for (int n = 1; n <= m * 2 - 1; n++) {
        System.out.print("*");
      }

      // Ending line after each row
      System.out.println();
    }
    }
}
