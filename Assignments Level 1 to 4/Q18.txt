import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
       Scanner scan = new Scanner(System.in);
      int n = scan.nextInt();
      for(int i=0; i<=n; i=i+2)
      {
      
          System.out.println(i);
     
      }
    }
}
