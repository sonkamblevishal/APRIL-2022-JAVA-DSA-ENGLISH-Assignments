//GKSTR06 Greater_Number

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
       Scanner scan = new Scanner(System.in);
      int a = scan.nextInt();
      int b = scan.nextInt();
      
      if(b>a)
      {
        System.out.println(b);
      }else{
        System.out.println(a);
      }
    }
}
